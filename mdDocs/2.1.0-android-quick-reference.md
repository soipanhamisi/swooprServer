### Create Application Class

```kotlin
class SwoopApplication : Application() {
    companion object {
        lateinit var apiService: SwoopApiService
        lateinit var tokenManager: TokenManager
        
        fun initializeApi(context: Context) {
            // Initialize token management
            tokenManager = TokenManager(context)
            
            // Build API client
            apiService = buildApiClient(context)
        }
        
        private fun buildApiClient(context: Context): SwoopApiService {
            val okHttpClient = OkHttpClient.Builder()
                .addInterceptor { chain ->
                    val originalRequest = chain.request()
                    val token = tokenManager.getToken()
                    
                    val newRequest = if (token != null) {
                        originalRequest.newBuilder()
                            .header("Authorization", "Bearer $token")
                            .build()
                    } else {
                        originalRequest
                    }
                    
                    chain.proceed(newRequest)
                }
                .addInterceptor(HttpLoggingInterceptor().apply {
                    level = HttpLoggingInterceptor.Level.BODY
                })
                .build()
            
            return Retrofit.Builder()
                .baseUrl(BuildConfig.API_BASE_URL)
                .client(okHttpClient)
                .addConverterFactory(GsonConverterFactory.create())
                .build()
                .create(SwoopApiService::class.java)
        }
    }
    
    override fun onCreate() {
        super.onCreate()
        initializeApi(this)
    }
}
```

## Common Code Snippets

### 1. Token Management

```kotlin
// Store token securely
class TokenManager(private val context: Context) {
    private val encryptedSharedPrefs = EncryptedSharedPreferences.create(
        context,
        "token_prefs",
        MasterKey.Builder(context).setKeyScheme(MasterKey.KeyScheme.AES256_GCM).build(),
        EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
        EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM
    )
    
    fun saveToken(token: String, expiresIn: Long) {
        val expirationTime = System.currentTimeMillis() + (expiresIn * 1000L)
        encryptedSharedPrefs.edit().apply {
            putString("jwt_token", token)
            putLong("token_expiration", expirationTime)
            apply()
        }
    }
    
    fun getToken(): String? {
        val token = encryptedSharedPrefs.getString("jwt_token", null)
        val expirationTime = encryptedSharedPrefs.getLong("token_expiration", 0L)
        
        return if (token != null && System.currentTimeMillis() < expirationTime) {
            token
        } else {
            null
        }
    }
    
    fun isTokenValid(): Boolean = getToken() != null
    
    fun clearToken() {
        encryptedSharedPrefs.edit().clear().apply()
    }
}
```

### 2. Authentication Flow

```kotlin
// Request OTP
suspend fun requestOtp(email: String): Result<Unit> {
    return try {
        val response = apiService.getOtp(EmailDTO(email))
        if (response.success) {
            Result.success(Unit)
        } else {
            Result.failure(Exception(response.message))
        }
    } catch (e: Exception) {
        Result.failure(e)
    }
}

// Verify OTP and get token
suspend fun verifyOtp(email: String, otp: Int): Result<String> {
    return try {
        val response = apiService.getNewToken(EmailAuthCredsDTO(email, otp))
        if (response.success) {
            val token = response.data?.token ?: return Result.failure(Exception("No token in response"))
            val expiresIn = response.data?.expiresIn ?: 604800L
            
            tokenManager.saveToken(token, expiresIn)
            Result.success(token)
        } else {
            Result.failure(Exception(response.message))
        }
    } catch (e: Exception) {
        Result.failure(e)
    }
}

// Register user
suspend fun registerUser(
    fullName: String,
    email: String,
    role: String,
    fcmToken: String
): Result<User> {
    return try {
        val dto = UserDTO(fullName, email, role, fcmToken)
        val response = apiService.registerUser(dto)
        
        if (response.success) {
            Result.success(response.data!!)
        } else {
            Result.failure(Exception(response.message))
        }
    } catch (e: Exception) {
        Result.failure(e)
    }
}
```

class SwoopMessagingService : FirebaseMessagingService() {
    override fun onMessageReceived(remoteMessage: RemoteMessage) {
        val title = remoteMessage.data["title"] ?: "Swoopd"
        val body = remoteMessage.data["body"] ?: ""
        val tripId = remoteMessage.data["tripId"]
        
        showNotification(title, body, tripId)
    }
    
    override fun onNewToken(token: String) {
        // Send new token to backend
        sendTokenToBackend(token)
    }
    
    private fun showNotification(title: String, body: String, tripId: String?) {
        val intent = Intent(this, MainActivity::class.java).apply {
            if (tripId != null) putExtra("tripId", tripId)
        }
        
        val pendingIntent = PendingIntent.getActivity(
            this, 0, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )
        
        val notification = NotificationCompat.Builder(this, "SWOOPD_TRIPS")
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setContentText(body)
            .setContentIntent(pendingIntent)
            .setAutoCancel(true)
            .build()
        
        NotificationManagerCompat.from(this).notify(1, notification)
    }
    
    private fun sendTokenToBackend(token: String) {
        // Send to backend using ApiService
    }
}

// In Activity/Fragment
override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    
    // Request notification permission (Android 13+)
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
        requestPermissions(

        }

### 6. ViewModel Pattern with LiveData

```kotlin
class TripViewModel(private val repository: TripRepository) : ViewModel() {
    private val _createTripState = MutableLiveData<Result<TripData>>()
    val createTripState: LiveData<Result<TripData>> = _createTripState
    
    fun createTrip(
        capacity: Int,
        departureTime: String,
        coords: Coordinates
    ) {
        viewModelScope.launch {
            _createTripState.value = Result.loading()
            
            val result = repository.createTrip(capacity, departureTime, coords)
            _createTripState.value = result
        }
    }
}

// Usage in Activity
class CreateTripActivity : AppCompatActivity() {
    private val viewModel: TripViewModel by viewModels()
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        viewModel.createTripState.observe(this) { result ->
            when (result) {
                is Result.Loading -> showLoadingIndicator()
                is Result.Success -> {
                    hideLoadingIndicator()
                    navigateToTripDetails(result.data)
                }
                is Result.Error -> {
                    hideLoadingIndicator()
                    showErrorMessage(result.exception.message)
                }
            }
        }
    }
}
```

### 7. Error Handling

```kotlin
sealed class Result<T>(
    val data: T? = null,
    val error: Throwable? = null
) {
    class Loading<T> : Result<T>()
    class Success<T>(data: T) : Result<T>(data = data)
    class Error<T>(error: Throwable) : Result<T>(error = error)
    
    companion object {
        fun <T> loading(): Result<T> = Loading()
        fun <T> success(data: T): Result<T> = Success(data)
        fun <T> error(error: Throwable): Result<T> = Error(error)
    }
}

// Handle HTTP errors
fun handleApiError(throwable: Throwable): String {
    return when (throwable) {
        is HttpException -> {
            when (throwable.code()) {
                401 -> "Authentication failed. Please login again."
                403 -> "You don't have permission for this action."
                404 -> "Resource not found."
                409 -> "This resource already exists."
                429 -> "Too many requests. Please try again later."
                500, 503 -> "Server error. Please try again later."
                else -> "An error occurred: ${throwable.message()}"
            }
        }
        is IOException -> "Network error. Please check your connection."
        else -> "Unknown error: ${throwable.message}"
    }
}
```

## Pro Tips

1. **Use BuildConfig for URLs:** Keep API URLs in build.gradle, not hardcoded
2. **Secure Token Storage:** Always use EncryptedSharedPreferences
3. **Test on Real Device:** Emulator behavior differs for location and notifications
4. **Firebase Testing:** Use Firebase Console to send test notifications
5. **Retrofit Logging:** Enable HttpLoggingInterceptor only in debug builds
6. **Handle Backlog:** App should show "Waiting for match" state clearly
7. **Implement Retry:** Network calls should retry on 5xx errors
8. **Monitor Token Expiry:** Refresh token 1 day before expiration
9. **Test Permissions:** Request location and notification permissions properly
10. **Error Messages:** Show user-friendly messages, log technical details
